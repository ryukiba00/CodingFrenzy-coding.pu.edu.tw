import java.util.*;

class main{
	public static void main(String args[]){
	Scanner sc = new Scanner(System.in);
	
	int m = sc.nextInt(); // 俱计mA皑降{ (そへ) 计
	int n = 80; // 俱计nA盎荪Ió戈C
	
	if ( m == 0 ){ // 降{计0
		n = 0;
	}else if ( m <= 1500 ){ // 降{计b1500そへずO80じ
		n = 80;
	}else{ // WL1500そへ
		n += 5*((m-1500)/500); // WL1500そへ
		m = (m-1500)%500;
		if ( m > 0 ){ // pGぃì500そへH500そへp
			n += 5;
		}
	}
	System.out.println(n);
	}
}
